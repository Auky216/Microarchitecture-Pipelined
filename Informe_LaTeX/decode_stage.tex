% Extend macro
\newcommand{\drawextend}[3]{
  \draw[thick, fill=pastelblue] 
    (#1-1.2, #2+0.3) -- 
    (#1+1.2, #2+0.6) -- 
    (#1+1.2, #2-0.6) -- 
    (#1-1.2, #2-0.3) -- cycle;
  \node[font=\small\sffamily] (#3) at (#1, #2) {Extend};
  \coordinate (#3_in) at (#1-1.2, #2);
  \coordinate (#3_out) at (#1+1.2, #2);
}
